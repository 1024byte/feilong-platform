169      * 将日期集合转成特定pattern的字符串集合.
170      * 
171      * @param dateList
172      *            日期集合
173      * @param datePattern
174      *            模式 {@link DatePattern}
175      * @return 如果 <code>dateList</code> 是null或者empty,返回 {@link Collections#emptyList()}<br>
176      *         否则循环date转成string,返回{@code List<String>}
177      */
178     public static List<String> toStringList(List<Date> dateList,String datePattern){
179         if (isNullOrEmpty(dateList)){
180             return Collections.emptyList();
181         }
182 
183         Validate.notBlank(datePattern, "datePattern can't be blank!");
184 
185         List<String> stringList = new ArrayList<String>();
186         for (Date date : dateList){
187             stringList.add(DateUtil.toString(date, datePattern));
188         }
189         return stringList;
190     }

287         Validate.notBlank(datePattern, "datePattern can't be blank!");
288         Date now = new Date();
289         Date firstWeekOfSpecifyDateYear = DateUtil.getFirstWeekOfSpecifyDateYear(now, week);
290         //当年最后一天
291         Calendar calendarEnd = CalendarUtil.resetYearEnd(toCalendar(now));
292 
293         List<String> list = new ArrayList<String>();
294         Calendar firstWeekOfSpecifyDateYearCalendar = toCalendar(firstWeekOfSpecifyDateYear);
295         for (Calendar calendar = firstWeekOfSpecifyDateYearCalendar; calendar.before(calendarEnd); calendar.add(Calendar.DAY_OF_YEAR, 7)){
296             list.add(CalendarUtil.toString(calendar, datePattern));
297         }
298         return list;
299     }

303     /**
304      * 人性化显示date时间,依据是和现在的时间间隔.
305      * 
306      * <p>
307      * 转换规则,将传入的inDate和 new Date()当前时间比较;当两者的时间差,(一般inDate小于当前时间 ,暂时不支持大于当前时间)
308      * </p>
309      * 
310      * <ul>
311      * <li>如果时间差为0天,<br>
312      * 如果小时间隔等于0,如果分钟间隔为0,则显示间隔秒 + "秒钟前"<br>
313      * 如果小时间隔等于0,如果分钟间隔不为0,则显示间隔分钟 + "分钟前"<br>
314      * </li>
315      * 
316      * <li>如果时间差为0天,<br>
317      * 如果小时间隔不等于0,如果inDate的day 和current的day 相等,则显示space_hour + "小时前"<br>
318      * 如果小时间隔不等于0,如果inDate的day 和current的day不相等,则显示"昨天 " + toString(inDate, "HH:mm")<br>
319      * </li>
320      * 
321      * <li>如果时间差为1天,且inDate的day+1和currentDate的day 相等,则显示"昨天 HH:mm"</li>
322      * <li>如果时间差为1天,且inDate的day+1和currentDate的day 不相等,则显示"前天 HH:mm"</li>
323      * <li>如果时间差为2天,且inDate的day+2和currentDate的day 相等,则显示"前天 HH:mm"</li>
324      * <li>如果时间差为2天,且inDate的day+2和currentDate的day 不相等,<br>
325      * 1).如果inDate的year和currentDate的year相等,则显示"MM-dd HH:mm"<br>
326      * 2).如果inDate的year和currentDate的year不相等,则显示"yyyy-MM-dd HH:mm"
327      * </li>
328      * 
329      * <li>如果时间差大于2天<br>
330      * 1).如果inDate的year和currentDate的year相等,则显示"MM-dd HH:mm"<br>
331      * 2).如果inDate的year和currentDate的year不相等,则显示"yyyy-MM-dd HH:mm"
332      * </li>
333      * 
334      * </ul>
335      * 
336      * @param inDate
337      *            任意日期<br>
338      *            warn:一般inDate{@code <=}当前时间 ,暂时不支持大于当前时间
339      * @return 人性化显示date时间<br>
340      *         如果 <code>inDate</code> 是null,抛出 {@link NullPointerException}<br>
341      * @see DateUtil#toString(Date, String)
342      * @see DateUtil#getYear(Date)
343      * @see DateUtil#getDayOfMonth(Date)
344      * @see DateUtil#getYear(Date)
345      * @see DateExtensionUtil#getIntervalTime(Date, Date)
346      * @see DateExtensionUtil#getIntervalDay(long)
347      * @see DateExtensionUtil#getIntervalHour(long)
348      * @see DateExtensionUtil#getIntervalMinute(long)
349      * @see DateExtensionUtil#getIntervalSecond(long)
350      */
351     public static String toPrettyDateString(Date inDate){
352         Validate.notNull(inDate, "inDate can't be null!");
353 
354         Date nowDate = new Date();
355 
356         // 传过来的日期的年份
357         int inYear = DateUtil.getYear(inDate);
358         //**************************************************************************************/
359         int currentYear = DateUtil.getYear(nowDate);// 当前时间的年
360         boolean isSameYear = currentYear == inYear;//是否是同一年
361         long spaceTime = DateExtensionUtil.getIntervalTime(inDate, nowDate);// 任意日期和现在相差的毫秒数
362         int spaceDay = DateExtensionUtil.getIntervalDay(spaceTime);// 相差天数
363         //**************************************************************************************/
364         switch (spaceDay) {
365             case 0: // 间隔0天
366                 return doWithZeroDayInterval(inDate, nowDate, spaceTime);
367             case 1: // 间隔一天
368                 return doWithOneDayInterval(inDate, nowDate);
369             case 2: // 间隔2天
370                 return doWithTwoDaysInterval(inDate, nowDate, isSameYear);
371             default://spaceDay > 2     // 间隔大于2天
372                 return isSameYear ? DateUtil.toString(inDate, COMMON_DATE_AND_TIME_WITHOUT_YEAR_AND_SECOND)
373                                 : DateUtil.toString(inDate, COMMON_DATE_AND_TIME_WITHOUT_SECOND);
374         }
375     }

381     /**
382      * Do with one day interval.
383      *
384      * @param inDate
385      *            the in date
386      * @param nowDate
387      *            the now date
388      * @return the string
389      * @since 1.4.0
390      */
391     private static String doWithOneDayInterval(Date inDate,Date nowDate){
392         return DateUtil.isEquals(DateUtil.addDay(inDate, 1), nowDate, COMMON_DATE)
393                         ? YESTERDAY + " " + DateUtil.toString(inDate, COMMON_TIME_WITHOUT_SECOND)
394                         : THEDAY_BEFORE_YESTERDAY + " " + DateUtil.toString(inDate, COMMON_TIME_WITHOUT_SECOND);
395     }

397     /**
398      * Do with two days interval.
399      *
400      * @param inDate
401      *            the in date
402      * @param nowDate
403      *            the now date
404      * @param isSameYear
405      *            the is same year
406      * @return the string
407      * @since 1.4.0
408      */
409     private static String doWithTwoDaysInterval(Date inDate,Date nowDate,boolean isSameYear){
410         if (DateUtil.isEquals(DateUtil.addDay(inDate, 2), nowDate, COMMON_DATE)){
411             return THEDAY_BEFORE_YESTERDAY + " " + DateUtil.toString(inDate, COMMON_TIME_WITHOUT_SECOND);
412         }
413         return isSameYear ? DateUtil.toString(inDate, COMMON_DATE_AND_TIME_WITHOUT_YEAR_AND_SECOND)
414                         : DateUtil.toString(inDate, COMMON_DATE_AND_TIME_WITHOUT_SECOND);
415     }

417     /**
418      * Do with zero day interval.
419      *
420      * @param inDate
421      *            the in date
422      * @param nowDate
423      *            the now date
424      * @param spaceTime
425      *            the space time
426      * @return the string
427      * @since 1.4.0
428      */
429     private static String doWithZeroDayInterval(Date inDate,Date nowDate,long spaceTime){
430         int spaceHour = DateExtensionUtil.getIntervalHour(spaceTime); // 相差小时数
431         if (spaceHour == 0){// 小时间隔
432             int spaceMinute = DateExtensionUtil.getIntervalMinute(spaceTime);
433             return spaceMinute == 0 ? DateExtensionUtil.getIntervalSecond(spaceTime) + SECOND + "前" : spaceMinute + MINUTE + "前";
434         }
435         // 传过来的日期的日
436         int inDay = DateUtil.getDayOfMonth(inDate);
437         // 当前时间的日
438         int currentDayOfMonth = DateUtil.getDayOfMonth(nowDate);
439         return inDay == currentDayOfMonth ? spaceHour + HOUR + "前"
440                         : YESTERDAY + " " + DateUtil.toString(inDate, COMMON_TIME_WITHOUT_SECOND);
441     }

443     /**
444      * 获得两个日期时间的日期间隔时间集合(包含最小和最大值),用于统计日报表.
445      * 
446      * <h3>示例:</h3>
447      * <blockquote>
448      * 
449      * <pre class="code">
450      * Date fromDate = DateUtil.toDate("2011-03-5 23:31:25.456", DatePattern.COMMON_DATE_AND_TIME);
451      * Date toDate = DateUtil.toDate("2011-03-10 01:30:24.895", DatePattern.COMMON_DATE_AND_TIME);
452      * LOGGER.debug(JsonUtil.format(DateExtensionUtil.getIntervalDayList(fromDate, toDate)));
453      * </pre>
454      * 
455      * <b>返回:</b>
456      * 
457      * <pre class="code">
458      * ["2011-03-05 00:00:00",
459      * "2011-03-06 00:00:00",
460      * "2011-03-07 00:00:00",
461      * "2011-03-08 00:00:00",
462      * "2011-03-09 00:00:00",
463      * "2011-03-10 00:00:00"
464      * ]
465      * </pre>
466      * 
467      * </blockquote>
468      * 
469      * <h3>说明:</h3>
470      * <blockquote>
471      * <ol>
472      * <li>每天的日期会被重置清零 <code>00:00:00.000</code></li>
473      * <li>方法自动辨识 <code>fromDate</code>和 <code>toDate</code>哪个是开始时间</li>
474      * </ol>
475      * </blockquote>
476      * 
477      * @param fromDate
478      *            the from date
479      * @param toDate
480      *            the to date
481      * @return 如果 <code>fromDate</code> 是null,抛出 {@link NullPointerException}<br>
482      *         如果 <code>toDate</code> 是null,抛出 {@link NullPointerException}
483      * @see #getIntervalDay(Date, Date)
484      * @see org.apache.commons.lang3.time.DateUtils#iterator(Calendar, int)
485      * @since 1.5.4
486      */
487     public static List<Date> getIntervalDayList(Date fromDate,Date toDate){
488         Validate.notNull(fromDate, "fromDate can't be null!");
489         Validate.notNull(toDate, "toDate can't be null!");
490 
491         Date minDate = fromDate.before(toDate) ? fromDate : toDate;
492         Date maxDate = fromDate.before(toDate) ? toDate : fromDate;
493 
494         // ******重置时间********
495         Date beginDateReset = DateUtil.getFirstDateOfThisDay(minDate);
496         Date endDateReset = DateUtil.getLastDateOfThisDay(maxDate);
497 
498         List<Date> list = new ArrayList<>();
499         list.add(beginDateReset);
500 
501         // 相隔的天数
502         int intervalDay = DateExtensionUtil.getIntervalDay(beginDateReset, endDateReset);
503         for (int i = 0; i < intervalDay; ++i){
504             list.add(DateUtil.addDay(beginDateReset, i + 1));
505         }
506 
507         return list;
508     }
