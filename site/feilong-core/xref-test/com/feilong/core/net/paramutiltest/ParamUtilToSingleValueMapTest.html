16  package com.feilong.core.net.paramutiltest;
17  
18  import static java.util.Collections.emptyMap;
19  import static org.hamcrest.Matchers.allOf;
20  import static org.hamcrest.Matchers.hasEntry;
21  import static org.junit.Assert.assertEquals;
22  import static org.junit.Assert.assertThat;
23  
24  import org.junit.Test;
25  
26  import com.feilong.core.net.ParamUtil;
27  
28  import static com.feilong.core.CharsetType.UTF8;
29  
30  public class ParamUtilToSingleValueMapTest{
31  
32      /**
33       * Parses the query to value map.
34       */
35      @Test
36      public void parseQueryToValueMap(){
37          assertThat(ParamUtil.toSingleValueMap("a=1&b=2", UTF8), allOf(//
38                          hasEntry("a", "1"),
39                          hasEntry("b", "2")));
40  
41          assertThat(ParamUtil.toSingleValueMap("a=", UTF8), allOf(hasEntry("a", "")));
42          assertThat(ParamUtil.toSingleValueMap("a=1&", UTF8), allOf(hasEntry("a", "1")));
43      }
44  
45      @Test
46      public void testToSingleValueMapDuplicateParam(){
47          assertThat(ParamUtil.toSingleValueMap("a=1&b=2&a=3", UTF8), allOf(//
48                          hasEntry("a", "1"),
49                          hasEntry("b", "2")));
50      }
51  
52      @Test
53      public void testToSingleValueMapNullQueryString(){
54          assertEquals(emptyMap(), ParamUtil.toSingleValueMap(null, UTF8));
55      }
56  
57      @Test
58      public void testToSingleValueMapEmptyQueryString(){
59          assertEquals(emptyMap(), ParamUtil.toSingleValueMap("", UTF8));
60      }
61  
62      /**
63       * Test to single value map.
64       */
65      @Test
66      public void testToSingleValueMap(){
67          String queryString = "sec_id=MD5&format=xml&sign=cc945983476d615ca66cee41a883f6c1&v=2.0&req_data=%3Cauth_and_execute_req%3E%3Crequest_token%3E201511191eb5762bd0150ab33ed73976f7639893%3C%2Frequest_token%3E%3C%2Fauth_and_execute_req%3E&service=alipay.wap.auth.authAndExecute&partner=2088011438559510";
68          assertThat(ParamUtil.toSingleValueMap(queryString, UTF8), allOf(//
69                          hasEntry("sec_id", "MD5"),
70                          hasEntry("format", "xml"),
71                          hasEntry("sign", "cc945983476d615ca66cee41a883f6c1"),
72                          hasEntry("v", "2.0"),
73                          hasEntry(
74                                          "req_data",
75                                          "%3Cauth_and_execute_req%3E%3Crequest_token%3E201511191eb5762bd0150ab33ed73976f7639893%3C%2Frequest_token%3E%3C%2Fauth_and_execute_req%3E"),
76                          hasEntry("service", "alipay.wap.auth.authAndExecute"),
77                          hasEntry("partner", "2088011438559510")));
78      }
79  
80  }
