81      /**
82       * Test join values.
83       */
84      @Test
85      public void testJoinValues(){
86          String value = "create_salesorder";
87          String value2 = "unionpay_mobile";
88  
89          Map<String, String> map = new HashMap<String, String>();
90          map.put("service", value);
91          map.put("paymentType", value2);
92  
93          assertEquals(StringUtils.EMPTY, ParamUtil.joinValuesOrderByIncludeKeys(map, "a", "b"));
94          assertEquals(value, ParamUtil.joinValuesOrderByIncludeKeys(map, "service"));
95          assertEquals(value + value2, ParamUtil.joinValuesOrderByIncludeKeys(map, "service", "paymentType"));
96          assertEquals(value2 + value, ParamUtil.joinValuesOrderByIncludeKeys(map, "paymentType", "service"));
97      }
