200     /**
201      * Builds the message log.
202      *
203      * @param writeIndex
204      *            the write index
205      * @param bookSectionUrlMapSize
206      *            the book section url map size
207      * @param sectionName
208      *            the section name
209      * @param contentLength
210      *            the content length
211      * @return the string
212      */
213     private static String buildMessageLog(int writeIndex,int bookSectionUrlMapSize,String sectionName,int contentLength){
214         //进度,百分比
215         String progress = NumberUtil.getProgress(writeIndex + 1, bookSectionUrlMapSize);
216         String format = "%1$-20s %2$-6s %3$-3s";
217 
218         return StringUtil.format(format, sectionName, "" + contentLength, progress).replace(" ", "\u3000");
219         //String padStr = "-";
220         //return StringUtils.rightPad(sectionName, 40, padStr) + StringUtils.rightPad("" + contentLength, 6, padStr) + progress;
221     }
