57      /**
58       * 将数组转成转成 {@link java.util.Iterator}.
59       * <p>
60       * 如果我们幸运的话,它是一个对象数组,我们可以遍历并with no copying<br>
61       * 否则,异常 ClassCastException 中 ,Rats -- 它是一个基本类型数组,循环放入arrayList 转成arrayList.iterator()
62       * </p>
63       * <p>
64       * <b>注:</b>{@link Arrays#asList(Object...)} 转的list是 {@link Array} 的内部类 ArrayList,这个类没有实现
65       * {@link java.util.AbstractList#add(int, Object)} 这个方法,<br>
66       * 如果拿这个list进行add操作,会出现 {@link java.lang.UnsupportedOperationException}
67       * </p>
68       * 
69       * @param <T>
70       *            the generic type
71       * @param arrays
72       *            数组,可以是 对象数组,或者是 基本类型数组
73       * @return 如果 (null == arrays) 返回 null;<br>
74       *         否则会先将arrays转成Object[]数组,调用 {@link Arrays#asList(Object...)}转成list,再调用 {@link List#iterator()
75       *         t}<br>
76       *         对于基本类型的数组,由于不是 Object[],会有类型转换异常,此时先通过 {@link Array#getLength(Object)}取到数组长度,循环调用 {@link Array#get(Object, int)}设置到 list中
77       * @see Arrays#asList(Object...)
78       * @see Array#getLength(Object)
79       * @see Array#get(Object, int)
80       * @see List#iterator()
81       * @deprecated
82       */
83      @Deprecated
84      @SuppressWarnings({ "unchecked" })
85      public static <T> Iterator<T> toIterator(Object arrays){
86          if (null == arrays){
87              return null;
88          }
89          List<T> list = null;
90          try{
91              // 如果我们幸运的话,它是一个对象数组,我们可以遍历并with no copying
92              Object[] objArrays = (Object[]) arrays;
93              list = (List<T>) ConvertUtil.toList(objArrays);
94          }catch (ClassCastException e){
95              LOGGER.debug("arrays can not cast to Object[],maybe primitive type,values is:{},{}", arrays, e.getMessage());
96              // Rats -- 它是一个基本类型数组
97              int length = Array.getLength(arrays);
98              list = new ArrayList<T>(length);
99              for (int i = 0; i < length; ++i){
100                 Object object = Array.get(arrays, i);
101                 list.add((T) object);
102             }
103         }
104         return list.iterator();
105     }
