297     @Test
298     public void testGroup2(){
299         User zhangfei28 = new User("张飞", 28);
300         User liubei32 = new User("刘备", 32);
301         User liubei30 = new User("刘备", 30);
302         List<User> list = toList(//
303                         new User("张飞", 10),
304                         zhangfei28,
305                         liubei32,
306                         liubei30,
307                         new User("刘备", 10));
308 
309         Predicate<User> comparatorPredicate = BeanPredicateUtil.comparatorPredicate("age", 20, Criterion.LESS);
310         Map<String, List<User>> map = CollectionsUtil.group(list, "name", comparatorPredicate);
311         LOGGER.debug(JsonUtil.format(map));
312 
313         assertThat(map, allOf(//
314                         hasKey("张飞"),
315                         hasKey("刘备"),
316                         hasEntry(is("张飞"), hasItem(zhangfei28)),
317                         hasEntry(is("刘备"), hasItems(liubei32, liubei30))));
318         assertSame(2, map.size());
319 
320     }
