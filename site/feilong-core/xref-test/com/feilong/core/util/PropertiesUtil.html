27  /**
28   * 操作{@link Properties}配置文件.
29   * 
30   * <h3>关于{@link "Properties#load0(LineReader)"}核心方法:</h3>
31   * 
32   * <blockquote>
33   * <ol>
34   * <li>注释符为:'#'或者'!'.<br>
35   * 空白字符为:' ', '\t', '\f'.<br>
36   * key/value分隔符为:'='或者':'.<br>
37   * 行分隔符为:'\r','\n','\r\n'. </li>
38   * <li>自然行是使用行分隔符或者流的结尾来分割的行.<br>
39   * 逻辑行可能分割到多个自然行中,使用反斜杠'\'来连接多个自然行;</li>
40   * <li>注释行是使用注释符作为首个非空白字符的自然行;</li>
41   * <li>空白字符的自然行会被认为是空行而被忽略;</li>
42   * <li>key为<span style="color:red">从首个非空白字符开始</span>直到(但不包括)首个非转义的'=', ':'或者非行结束符的空白字符为止;</li>
43   * <li>key后面的第一个非空白字符如果是"="或者":",那么这个字符后面所有空白字符都会被忽略掉;</li>
44   * <li>可以使用转义序列表示key和value;</li>
45   * <li>value的界定为:逻辑行中,非转义的key/value分隔符(此处不仅仅包括'=',':',还包括' ', '\t', '\f')后面的<span style="color:red">第一个非空白字符(非' ', '\t', '\f'字符)</span>
46   * 开始到逻辑行结束的所有字符;</li>
47   * <li>如果空白字符后续没有key/value分隔符("="或者":"),<br>
48   * 那么该空白字符会被当作key/value分隔符,从分隔符后的第一个非空白字符起到逻辑行结束所有的字符都当作是value.<br>
49   * 也就是说:"key1 value1",读取出来之后的key和value分别为"key1", "value1". 
50   * </li>
51   * <li>如果空白字符后续有key/value分隔符("="或者":"),<br>
52   * 那么该空白字符会被忽略,key/value分隔符后的第一个非空白字符起到逻辑行结束所有的字符都当作是value.<br>
53   * 也就是说:"key1 :value1",读取出来之后的key和value分别为"key1"和"value1",而不是"key1"和":value1". 
54   * </li>
55   * </ol>
56   * </blockquote>
57   * 
58   * <h3>关于properties文件 value 换行:</h3>
59   * 
60   * <blockquote>
61   * 
62   * <p>
63   * 我们经常在properties文件中设置属性的时候,如果某一个属性的值太长,那么查看就不太方便,但是又不能直接的换行,否则读取属性的值的时候其换行部分就被忽略了.
64   * 其实我们可以通过增加一个\符号来达到换行的效果.如下:
65   * </p>
66   * 
67   * <pre class="code">
68      name=Hello world \
69      My Name is ferreousbox
70   * </pre>
71   * 
72   * <p>
73   * 那么我们在读取name属性的时其值就变成了:Hello world My Name is ferreousbox.也就解决了在properties文件中换行书写的问题,
74   * 只需要在每一行的最后增加一个\符号,注意其下一行必须是接着的一行,即中间不能空行,否则也会被忽略的:-).
75   * </p>
76   * 
77   * </blockquote>
78   *
79   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
80   * @see "org.apache.velocity.texen.util.PropertiesUtil"
81   * @see "org.apache.cxf.common.util.PropertiesLoaderUtils"
82   * @see "org.springframework.core.io.support.PropertiesLoaderUtils"
83   * @see "org.springframework.core.io.support.PropertiesLoaderSupport"
84   * @see "org.apache.commons.configuration.PropertiesConfiguration"
85   * @since 1.4.0
86   */
87  public final class PropertiesUtil{
88  
89      /** Don't let anyone instantiate this class. */
90      private PropertiesUtil(){
91          //AssertionError不是必须的. 但它可以避免不小心在类的内部调用构造器. 保证该类在任何情况下都不会被实例化.
92          //see 《Effective Java》 2nd
93          throw new AssertionError("No " + getClass().getName() + " instances for you!");
94      }

98      /**
99       * klass调用ClassLoader,通过ClassLoader 获取Properties.
100      * 
101      * <h3>关于路径 <code>propertiesPath</code>:</h3>
102      * <blockquote>
103      * 假设 class是 {@link PropertiesUtil},而配置文件在 src/main/resources下面,比如 messages/feilong-core-message_en_US.properties<br>
104      * 
105      * <p>
106      * 需要写作
107      * <code>getPropertiesWithClassLoader(PropertiesUtil.class,<b>"messages/feilong-core-message_en_US.properties"</b>)</code>
108      * <br>
109      * 注意此处要写成 <b>"messages/feilong-core-message_en_US.properties"</b> (不需要"/"),ClassLoader
110      * JVM会使用BootstrapLoader去加载资源文件.所以路径还是这种相对于工程的根目录
111      * </p>
112      * 
113      * <p>
114      * 注意和 {@link #getProperties(Class, String)} 的区别
115      * </p>
116      * </blockquote>
117      *
118      * @param klass
119      *            通过 klass 获得 ClassLoader,然后获得 getResourceAsStream
120      * @param propertiesPath
121      *            the properties path
122      * @return 如果 <code>klass</code> 是null,抛出 {@link NullPointerException}<br>
123      *         如果 <code>propertiesPath</code> 是null,抛出 {@link NullPointerException}<br>
124      *         如果 <code>propertiesPath</code> 是blank,抛出 {@link IllegalArgumentException}<br>
125      * @see ClassLoaderUtil#getClassLoaderByClass(Class)
126      * @see java.lang.ClassLoader#getResourceAsStream(String)
127      * @see #getProperties(InputStream)
128      * @see #getProperties(Class, String)
129      */
130     public static Properties getPropertiesWithClassLoader(Class<?> klass,String propertiesPath){
131         Validate.notBlank(propertiesPath, "propertiesPath can't be blank!");
132         ClassLoader classLoader = null;
133 
134         //ClassLoaderUtil.getClassLoaderByClass(klass);
135         return getProperties(classLoader.getResourceAsStream(propertiesPath));
136     }
137 
138     /**
139      * 通过{@link Class#getResourceAsStream(String)} 方法获得 InputStream,再调用 {@link #getProperties(InputStream)}.
140      * 
141      * <h3>关于路径 <code>propertiesPath</code>:</h3>
142      * 
143      * <blockquote>
144      * 假设 class是 {@link PropertiesUtil},而配置文件在 src/main/resources下面,比如 messages/feilong-core-message_en_US.properties<br>
145      * <p>
146      * 需要写作 <code>getProperties(PropertiesUtil.class,<b>"/messages/feilong-core-message_en_US.properties"</b>)</code> <br>
147      * 注意此处要写成 <b>"/messages/feilong-core-message_en_US.properties"</b>, 路径可以写成相对路径或者绝对路径
148      * </p>
149      * 
150      * <p>
151      * 注意和 {@link #getPropertiesWithClassLoader(Class, String)} 的区别
152      * </p>
153      * </blockquote>
154      * 
155      * @param klass
156      *            类,会通过 klass 调用
157      * @param propertiesPath
158      *            the properties path
159      * @return 如果 <code>klass</code> 是null,抛出 {@link NullPointerException}<br>
160      *         如果 <code>propertiesPath</code> 是null,抛出 {@link NullPointerException}<br>
161      *         如果 <code>propertiesPath</code> 是blank,抛出 {@link IllegalArgumentException}<br>
162      * @see java.lang.Class#getResourceAsStream(String)
163      * @see #getProperties(InputStream)
164      * @see #getPropertiesWithClassLoader(Class, String)
165      */
166     public static Properties getProperties(Class<?> klass,String propertiesPath){
167         Validate.notNull(klass, "klass can't be null!");
168         Validate.notBlank(propertiesPath, "propertiesPath can't be blank!");
169         // klass.getResourceAsStream方法 内部会调用classLoader.getResourceAsStream
170         // 之所以这样做无疑还是方便客户端的调用,省的每次获取ClassLoader才能加载资源文件的麻烦.
171         return getProperties(klass.getResourceAsStream(propertiesPath));
172     }
173 
174     /**
175      * 获取{@link Properties}.
176      *
177      * @param inputStream
178      *            inputStream
179      * @return 如果 <code>inputStream</code> 是null,抛出 {@link NullPointerException}<br>
180      *         正常情况,返回 {@link java.util.Properties#load(InputStream)}
181      * @see java.util.Properties#load(InputStream)
182      * @see "org.springframework.core.io.support.PropertiesLoaderUtils#loadProperties(Resource)"
183      */
184     public static Properties getProperties(InputStream inputStream){
185         Validate.notNull(inputStream, "inputStream can't be null!");
186         Properties properties = new Properties();
187         try{
188             properties.load(inputStream);
189             return properties;
190         }catch (IOException e){
191             throw new UncheckedIOException(e);
192         }
193     }
194     // [end]
195 }
