49      /**
50       * Read prefix as map({@link TreeMap}).
51       * 
52       * <p>
53       * 注意:JDK实现{@link java.util.PropertyResourceBundle},内部是使用 hashmap来存储数据的,<br>
54       * 本方法出于log以及使用方便,返回的是<span style="color:red"> TreeMap</span>
55       * </p>
56       * 
57       * <h3>示例:</h3>
58       * <blockquote>
59       * 
60       * 比如 messages/feilong-core-test.properties 里面有
61       * 
62       * <pre class="code">
63       * FileType.image=Image
64       * FileType.video=Video
65       * FileType.audio=Audio
66       * </pre>
67       * 
68       * 配置,此时调用
69       * 
70       * <pre class="code">
71       * Map{@code <String, String>} map = ResourceBundleUtil.readPrefixAsMap(BASE_NAME, "FileType", ".", Locale.CHINA);
72       * LOGGER.info(JsonUtil.format(map));
73       * </pre>
74       * 
75       * 返回 :
76       * 
77       * <pre class="code">
78       * {
79       * "audio": "Audio",
80       * "image": "Image",
81       * "video": "Video"
82       * }
83       * </pre>
84       * 
85       * </blockquote>
86       *
87       * @param baseName
88       *            一个完全限定类名,<b>配置文件的包+类全名</b>,比如 <b>message.feilong-core-test</b> <span style="color:red">(不要尾缀)</span>;<br>
89       *            但是,为了和早期版本兼容,也可使用路径名来访问,比如<b>message/feilong-core-test</b><span style="color:red">(使用 "/")</span>
90       * @param prefix
91       *            前缀
92       * @param delimiters
93       *            the delimiters
94       * @param locale
95       *            the locale for which a resource bundle is desired,如果是null,将使用 {@link Locale#getDefault()}
96       * @return 如果 <code>baseName</code> 是null,抛出 {@link NullPointerException}<br>
97       *         如果 <code>baseName</code> 是blank,抛出 {@link IllegalArgumentException}<br>
98       *         如果资源文件 <code>baseName</code> 不存在,抛出 <code>MissingResourceException</code><br>
99       *         如果 <code>baseName</code> 没有key value,则返回{@link java.util.Collections#emptyMap()}<br>
100      *         否则解析所有的key和value转成HashMap
101      * @see #readAllPropertiesToMap(String, Locale)
102      */
103     public static Map<String, String> readPrefixAsMap(String baseName,String prefix,String delimiters,Locale locale){
104         Map<String, String> propertyMap = ResourceBundleUtil.readAllPropertiesToMap(baseName, locale);
105         if (Validator.isNullOrEmpty(propertyMap)){
106             return Collections.emptyMap();
107         }
108 
109         Map<String, String> result = new TreeMap<String, String>();
110         for (Map.Entry<String, String> entry : propertyMap.entrySet()){
111             String key = entry.getKey();
112             // 以 prefix 开头
113             if (key.startsWith(prefix)){
114                 String[] values = StringUtil.tokenizeToStringArray(key, delimiters);
115                 if (values.length >= 2){
116                     result.put(values[1], entry.getValue());
117                 }
118             }
119         }
120         return result;
121     }

126     @Test

128         Map<String, String> map = readPrefixAsMap(BASE_NAME, "FileType", ".", Locale.CHINA);
129         LOGGER.debug(JsonUtil.format(map));
130     }
