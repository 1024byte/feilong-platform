69      /**
70       * 返回一个<code>bean</code>中所有的可读属性(read method),并将属性名/属性值放入一个Map中.
71       * 
72       * <h3>示例:</h3>
73       * <blockquote>
74       * 
75       * <pre class="code">
76       * User user = new User();
77       * user.setId(5L);
78       * user.setDate(new Date());
79       * 
80       * LOGGER.info(JsonUtil.format(BeanUtil.describe(user)));
81       * </pre>
82       * 
83       * <b>返回:</b>
84       * 
85       * <pre class="code">
86       * {
87       * "id": "5",
88       * "date": "Mon Apr 11 00:37:56 CST 2016"
89       * }
90       * </pre>
91       * 
92       * </blockquote>
93       * 
94       * <p>
95       * 另外还有一个名为class的属性,属性值是Object的类名,事实上class是java.lang.Object的一个属性.
96       * </p>
97       * 
98       * <p>
99       * <span style="color:red">缺陷:<br>
100      * 自己手工注册的ConvertUtils.register(dateTimeConverter, java.util.Date.class)不会生效</span><br>
101      * 
102      * 在赋值的时候,虽然调用了 {@link BeanUtilsBean#getNestedProperty(Object, String)}, 虽然也调用了 ConvertUtilsBean来转换 <br>
103      * 但是 {@link ConvertUtilsBean#ConvertUtilsBean()} 默认的构造函数 是使用标准的转换
104      * </p>
105      *
106      * @param bean
107      *            Bean whose properties are to be extracted
108      * @return 如果 <code>bean</code> 是null,返回 empty HashMap,see {@link BeanUtilsBean#describe(Object)}
109      * @throws BeanUtilException
110      *             有任何异常,转成{@link BeanUtilException}返回
111      * @see org.apache.commons.beanutils.BeanUtils#describe(Object)
112      * @see org.apache.commons.beanutils.PropertyUtils#describe(Object)
113      * @see PropertyUtil#describe(Object, String...)
114      * @see PropertyDescriptor
115      * @see BeanUtil#populate(Object, Map)
116      */
117     public static Map<String, String> describe(Object bean){
118         try{
119             //Return the entire set of properties for which the specified bean provides a read method.
120             return BeanUtils.describe(bean);
121         }catch (Exception e){
122             throw new BeanUtilException(e);
123         }
124     }

128     /**
129      * 创建 dyna bean.
130      * 
131      * <h3>示例:</h3>
132      * 
133      * <blockquote>
134      * 
135      * <pre class="code">
136      * 
137      * Map{@code <String, Class<?>>} typeMap = new HashMap<>();
138      * typeMap.put("address", java.util.Map.class);
139      * typeMap.put("firstName", String.class);
140      * typeMap.put("lastName", String.class);
141      * 
142      * Map{@code <String, Object>} valueMap = new HashMap<>();
143      * valueMap.put("address", new HashMap());
144      * valueMap.put("firstName", "Fred");
145      * valueMap.put("lastName", "Flintstone");
146      * 
147      * DynaBean dynaBean = BeanUtil.newDynaBean(typeMap, valueMap);
148      * LOGGER.debug(JsonUtil.format(dynaBean));
149      * 
150      * </pre>
151      * 
152      * <b>返回:</b>
153      * 
154      * <pre class="code">
155      * {
156      *         "lastName": "Flintstone",
157      *         "address": {},
158      *         "firstName": "Fred"
159      *     }
160      * </pre>
161      * 
162      * </blockquote>
163      *
164      * @param typeMap
165      *            属性名称 和类型的 map
166      * @param valueMap
167      *            属性名称 和 值 的map
168      * @return the dyna bean
169      * @throws NullPointerException
170      *             如果 <code>typeMap </code> 或者 <code>valueMap</code> 是null
171      * @throws IllegalArgumentException
172      *             如果 <code>typeMap.size() != valueMap.size()</code>
173      * @since 1.8.1 change name
174      */
175     public static DynaBean newDynaBean(Map<String, Class<?>> typeMap,Map<String, Object> valueMap){
176         Validate.notNull(typeMap, "typeMap can't be null!");
177         Validate.notNull(valueMap, "valueMap can't be null!");
178         Validate.isTrue(typeMap.size() == valueMap.size(), "typeMap size:[%s] != valueMap size:[%s]", typeMap.size(), valueMap.size());
179 
180         //*********************************************************************************
181         try{
182             DynaClass dynaClass = new BasicDynaClass(null, null, toDynaPropertyArray(typeMap));
183 
184             DynaBean dynaBean = dynaClass.newInstance();
185             for (Map.Entry<String, Object> entry : valueMap.entrySet()){
186                 dynaBean.set(entry.getKey(), entry.getValue());
187             }
188             return dynaBean;
189         }catch (IllegalAccessException | InstantiationException e){
190             LOGGER.error("", e);
191             throw new BeanUtilException(e);
192         }
193     }

195     /**
196      * To dyna property array.
197      *
198      * @param typeMap
199      *            the type map
200      * @return the dyna property[]
201      * @since 1.8.0
202      */
203     private static DynaProperty[] toDynaPropertyArray(Map<String, Class<?>> typeMap){
204         DynaProperty[] dynaPropertys = new DynaProperty[typeMap.size()];
205         int i = 0;
206         for (Map.Entry<String, Class<?>> entry : typeMap.entrySet()){
207             dynaPropertys[i] = new DynaProperty(entry.getKey(), entry.getValue());
208             i++;
209         }
210         return dynaPropertys;
211     }
