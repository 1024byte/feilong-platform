147         // accessing the city of first address

149         String city = (String) PropertyUtils.getProperty(customer, "addresses[0].city");
150 
151         LOGGER.debug(StringUtils.join(new Object[] { "The city of customer ", name, "'s first address is ", city, "." }));
152

154         String zipPattern = "addresses[1].zipCode";
155         if (PropertyUtils.isWriteable(customer, zipPattern)){//PropertyUtils  
156             LOGGER.debug("Setting zipcode ...");
157             PropertyUtils.setProperty(customer, zipPattern, "200000");//PropertyUtils  
158         }
159         String zip = (String) PropertyUtils.getProperty(customer, zipPattern);//PropertyUtils  
160 
161         LOGGER.debug(StringUtils.join(new Object[] { "The zipcode of customer ", name, "'s second address is now ", zip, "." }));

342     public void cloneBean2(){
343         OrderLine orderLine = new OrderLine();
344         orderLine.setCount(8);
345         orderLine.setSalePrice(toBigDecimal(599));
346 
347         List<OrderLine> list = toList(orderLine);
348 
349         //*******************************************************************
350 
351         List<OrderLine> list1 = list;
352         List<OrderLine> copyList = new ArrayList<>();
353 
354         for (OrderLine orderLineTemp : list){
355             copyList.add(BeanUtil.cloneBean(orderLineTemp));
356         }
357         //PropertyUtil.copyProperties(copyList, list);
358         //************************************************************
359 
360         // List<OrderLine> cloneList = BeanUtil.cloneBean(list);
361 
362         //************************************************************
363 
364         String format = JsonUtil.format(list, ConvertUtil.toArray("MSRP"), 0, 0);
365         LOGGER.debug("the param format:{}", format);
366 
367         List<OrderLine> serializelist = (List<OrderLine>) SerializationUtils.clone((Serializable) list);
368 
369         //******************************************************************
370         for (OrderLine perOrderLine : list){
371             perOrderLine.setSalePrice(toBigDecimal(200));
372         }
373         //******************************************************************
374         List<OrderLine> jsonList = JsonUtil.toList(format, OrderLine.class);
375 
376         assertEquals(toBigDecimal(200), list1.get(0).getSalePrice());
377         //assertEquals(toBigDecimal(599), cloneList.get(0).getSalePrice());
378         assertEquals(toBigDecimal(599), serializelist.get(0).getSalePrice());
379         assertEquals(toBigDecimal(599), jsonList.get(0).getSalePrice());
380         assertEquals(toBigDecimal(599), copyList.get(0).getSalePrice());
