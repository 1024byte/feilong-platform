330     /**
331      * 取到 {@link System#getProperty(String)},转成 {@link java.util.TreeMap},以遍输出log的时候,会顺序显示.
332      * 
333      * @return the properties map for log
334      * @see System#getProperties()
335      * @see ConvertUtil#toMap(Properties)
336      * @see "org.springframework.core.env.AbstractEnvironment#getSystemProperties()"
337      */
338     public static Map<String, String> getPropertiesMapForLog(){
339         Properties properties = System.getProperties();
340         return new TreeMap<String, String>(ConvertUtil.toMap(properties));
341     }
342 
343     /**
344      * 取到 {@link System#getenv()},转成 {@link java.util.TreeMap},以遍输出log的时候,会顺序显示.
345      * 
346      * @return the env map for log
347      * @see System#getenv()
348      * @see "org.springframework.core.env.AbstractEnvironment#getSystemEnvironment()"
349      */
350     public static Map<String, String> getEnvMapForLog(){
351         return new TreeMap<String, String>(System.getenv());
352     }
353 
354     /**
355      * 设置 properties from map.
356      * 
357      * <p>
358      * 如果 <code>map</code> 是null或者empty,抛出 {@link NullPointerException}<br>
359      * </p>
360      *
361      * @param map
362      *            the properties from map
363      * @see java.lang.System#setProperty(String, String)
364      * @since 1.2.0
365      */
366     public static void setPropertiesFromMap(Map<String, String> map){
367         Validate.notEmpty(map, "map can't be null/empty!");
368         for (Map.Entry<String, String> entry : map.entrySet()){
369             System.setProperty(entry.getKey(), entry.getValue());
370         }
371     }
372 
373     /**
374      * 设置 properties from properties.
375      * 
376      * <p>
377      * 如果 <code>properties</code> 是null,抛出 {@link NullPointerException}<br>
378      * </p>
379      *
380      * @param properties
381      *            the properties from properties
382      * @see ConvertUtil#toMap(Properties)
383      * @see #setPropertiesFromMap(Map)
384      * @see java.lang.System#setProperties(Properties)
385      * @since 1.2.0
386      */
387     public static void setPropertiesFromProperties(Properties properties){
388         Validate.notNull(properties, "properties can't be null!");
389         setPropertiesFromMap(ConvertUtil.toMap(properties));
390     }
