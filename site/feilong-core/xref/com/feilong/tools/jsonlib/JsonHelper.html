96      /**
97       * 把实体Bean、Map对象、数组、列表集合转换成{@link JSON}.
98       * 
99       * <h3>说明:</h3>
100      * 
101      * <blockquote>
102      * <ol>
103      * <li>如果 <code>null==jsonConfig</code>,将使用 {@link #DEFAULT_JAVA_TO_JSON_CONFIG}</li>
104      * 
105      * <li>
106      * 
107      * <p>
108      * 以下类型将转成{@link JSONArray}:
109      * </p>
110      * 
111      * <ul>
112      * <li>如果是字符串,当是以"["符号开头,"]"符号结尾的时候</li>
113      * <li>数组 obj.getClass().isArray()||obj instanceof Object[]</li>
114      * <li>集合 obj instanceof Collection</li>
115      * <li>枚举 obj instanceof Enum</li>
116      * <li>迭代器 obj instanceof Iterator</li>
117      * </ul>
118      * 
119      * </li>
120      * <li>其他类型转成 {@link JSONObject}</li>
121      * </ol>
122      * </blockquote>
123      *
124      * @param obj
125      *            可以是数组,字符串,枚举,集合,map,Java bean,Iterator等类型,内部自动识别转成{@link JSONArray}还是{@link JSONObject}
126      * @param jsonConfig
127      *            the json config
128      * @return the jSON
129      * @see net.sf.json.JSONArray#fromObject(Object, JsonConfig)
130      * @see net.sf.json.JSONObject#fromObject(Object, JsonConfig)
131      * @see net.sf.json.util.JSONUtils#isArray(Object)
132      * @see java.lang.Class#isEnum()
133      * @see net.sf.json.JsonConfig#registerJsonValueProcessor(Class, JsonValueProcessor)
134      * @see org.apache.commons.collections4.IteratorUtils#toList(Iterator)
135      * @see org.apache.commons.collections4.IteratorUtils#toList(Iterator, int)
136      * @see net.sf.json.JSONSerializer#toJSON(Object)
137      */
138     static JSON toJSON(Object obj,JsonConfig jsonConfig){
139         JsonConfig useJsonConfig = defaultIfNull(jsonConfig, DEFAULT_JAVA_TO_JSON_CONFIG);
140         registerDefaultJsonValueProcessor(useJsonConfig);
141 
142         if (isNeedConvertToJSONArray(obj)){
143             Object arrayJsonObject = obj instanceof Iterator ? IteratorUtils.toList((Iterator<?>) obj) : obj;
144             return JsonHelper.toJSONArray(arrayJsonObject, useJsonConfig);
145         }
146         return JsonHelper.toJSONObject(obj, useJsonConfig);
147     }

149     /**
150      * 是否需要转成 {@link JSONArray}类型.
151      * 
152      * <h3>以下类型将转成{@link JSONArray}:</h3>
153      * <blockquote>
154      * <ol>
155      * <li>如果是字符串,当是以"["符号开头,"]"符号结尾的时候</li>
156      * <li>数组 obj.getClass().isArray()||obj instanceof Object[]</li>
157      * <li>集合 obj instanceof Collection</li>
158      * <li>枚举 obj instanceof Enum</li>
159      * <li>迭代器 obj instanceof Iterator</li>
160      * </ol>
161      * </blockquote>
162      *
163      * @param obj
164      *            the obj
165      * @return true, if is need convert to JSON array
166      * @see net.sf.json.JSONArray#_fromJSONTokener(net.sf.json.util.JSONTokener, JsonConfig)
167      * @see net.sf.json.util.JSONUtils#isArray(Object)
168      */
169     private static boolean isNeedConvertToJSONArray(Object obj){
170         if (obj instanceof String){
171             String str = (String) obj;
172             if (str.startsWith("[") && str.endsWith("]")){// [] 格式的字符串 
173                 return true;
174             }
175         }
176         return JSONUtils.isArray(obj) || //obj.getClass().isArray() || obj instanceof Collection || obj instanceof Object[]
177                         obj instanceof Enum || // obj.getClass().isEnum()这么写 null会报错// object' is an Enum. Use JSONArray instead
178                         obj instanceof Iterator;
179     }

241     /**
242      * Builds the java bean to json config.
243      *
244      * @param javaBean
245      *            the obj
246      * @return 取到该javabean的Field,解析是否有 {@link SensitiveWords}注解,如果有,那么添加 {@link SensitiveWordsJsonValueProcessor}
247      */
248     static JavaToJsonConfig buildJavaBeanToJsonConfig(Object javaBean){
249         List<Field> fieldsList = FieldUtils.getFieldsListWithAnnotation(javaBean.getClass(), SensitiveWords.class);
250         if (isNotNullOrEmpty(fieldsList)){
251             //敏感字段
252             Map<String, JsonValueProcessor> propertyNameAndJsonValueProcessorMap = new HashMap<>();
253             for (Field field : fieldsList){
254                 propertyNameAndJsonValueProcessorMap.put(field.getName(), SensitiveWordsJsonValueProcessor.INSTANCE);
255             }
256             return new JavaToJsonConfig(propertyNameAndJsonValueProcessorMap);
257         }
258         return null;
259     }

262     /**
263      * 默认的java to json JsonConfig.
264      * 
265      * <h3>含有以下的特性:</h3>
266      * <blockquote>
267      * <ol>
268      * <li>{@link CycleDetectionStrategy#LENIENT} 避免循环引用</li>
269      * <li>no IgnoreDefaultExcludes,默认过滤几个key "class", "declaringClass","metaClass"</li>
270      * <li>
271      * {@link DateJsonValueProcessor},如果是日期,自动渲染成 {@link DatePattern#COMMON_DATE_AND_TIME} 格式类型,如有需要可以使用
272      * {@link JavaToJsonConfig#setPropertyNameAndJsonValueProcessorMap(Map)}覆盖此属性
273      * </li>
274      * <li>AllowNonStringKeys,允许非 string类型的key</li>
275      * </ol>
276      * </blockquote>
277      *
278      * @return the default json config
279      * @see see net.sf.json.JsonConfig#DEFAULT_EXCLUDES
280      * @see net.sf.json.util.CycleDetectionStrategy#LENIENT
281      * 
282      * @see <a href="http://feitianbenyue.iteye.com/blog/2046877">通过setAllowNonStringKeys解决java.lang.ClassCastException: JSON keys must be
283      *      strings</a>
284      */
285     static JsonConfig buildDefaultJavaToJsonConfig(){
286         JsonConfig jsonConfig = new JsonConfig();
287 
288         //see net.sf.json.JsonConfig#DEFAULT_EXCLUDES
289         //默认会过滤的几个key "class", "declaringClass","metaClass"  
290         jsonConfig.setIgnoreDefaultExcludes(false);
291 
292         // java.lang.ClassCastException: JSON keys must be strings
293         // see http://feitianbenyue.iteye.com/blog/2046877
294         jsonConfig.setAllowNonStringKeys(true);
295 
296         //排除,避免循环引用 There is a cycle in the hierarchy!
297         //Returns empty array and null object
298         jsonConfig.setCycleDetectionStrategy(CycleDetectionStrategy.LENIENT);
299 
300         // 注册日期处理器
301         jsonConfig.registerJsonValueProcessor(Date.class, DateJsonValueProcessor.DEFAULT_INSTANCE);
302         return jsonConfig;
303     }

305     /**
306      * 是否允许被json format的类型.
307      * 
308      * <h3>如果是以下类型是被允许的:</h3>
309      * <blockquote>
310      * <ol>
311      * <li>{@link ClassUtils#isPrimitiveOrWrapper(Class)}</li>
312      * <li>{@link String}</li>
313      * <li>{@link ObjectUtil#isArray(Object)}</li>
314      * <li>{@link ClassUtil#isInstanceAnyClass(Object, Class...)}</li>
315      * </ol>
316      * </blockquote>
317      *
318      * @param <V>
319      *            the value type
320      * @param value
321      *            the value
322      * @param allowClassTypes
323      *            the allow class types
324      * @return true, if checks if is allow format type
325      */
326     static <V> boolean isAllowFormatType(V value,Class<?>...allowClassTypes){
327         if (null == value){//null 是可以 format的
328             return true;
329         }
330         Class<?> klassClass = value.getClass();
331         return ClassUtils.isPrimitiveOrWrapper(klassClass) //
332                         || String.class == klassClass //
333                         || ObjectUtil.isArray(value)//XXX 数组一般 是可以转换的 
334                         || ClassUtil.isInstanceAnyClass(value, allowClassTypes)//
335         ;
336     }
